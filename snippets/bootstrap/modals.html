JavaScript usage

Hello there !
<script type="text/javascript">
    $(function() {
        $('#js-modal-trigger').click(function() {
            $('#js-modal').modal({
                backdrop: true, // (true|false|'static') Includes a modal-backdrop element.
                                // Alternatively, specify 'static' for a backdrop which
                                // doesn't close the modal on click.
                keyboard: true, // (true|false) Closes the modal when escape key is pressed.
                focus: true,    // (true|false) Puts the focus on the modal when initialized.
                show: true      // (true|false) Shows the modal when initialized.
            });
        });

        $('#js-modal-close').click(function() {
            $('#js-modal').modal('hide');
        });
    });
</script>

<button id="js-modal-trigger" class="btn btn-primary">Open modal</button>

<div id="js-modal" class="modal fade">
    <div class="modal-dialog modal-sm">
        <div class="modal-content">
            <div class="modal-body">Hello there !</div>
            <div class="modal-footer">
                <button id="js-modal-close" class="btn btn-secondary">Close</button>
                <button class="btn btn-primary">Save changes</button>
            </div>
        </div>
    </div>
</div>
